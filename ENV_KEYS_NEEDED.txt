━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  1. OPENROUTER  (Required — powers all AI features)
     Get it free at: https://openrouter.ai
     → Sign up → Keys → Create Key
     One key gives access to Claude, GPT-4o, Gemini, Llama, etc.
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  OPENROUTER_API_KEY = sk-or-v1-...
  OPENROUTER_MODEL   = anthropic/claude-sonnet-4-5

  Other model options (pick any one):
    anthropic/claude-opus-4-5
    openai/gpt-4o
    google/gemini-2.5-flash
    meta-llama/llama-3.3-70b-instruct


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  2. GOOGLE (Gemini — video analysis)
     Get it at: https://aistudio.google.com
     → Get API Key
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  GOOGLE_API_KEY = AIza...


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  3. PINECONE (Vector DB — RAG-powered script generation)
     Get it at: https://app.pinecone.io
     → Create index → API Keys
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  PINECONE_API_KEY    = pcsk_...
  PINECONE_INDEX_NAME = instagram-content-patterns
  PINECONE_ENVIRONMENT = us-east-1


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  4. META / INSTAGRAM (Connect user's own IG account)
     Get it at: https://developers.facebook.com
     → My Apps → Create App → Business type
     → Add Instagram Graph API product
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  META_APP_ID     = (numbers, e.g. 123456789)
  META_APP_SECRET = (hex string)

  ⚠ Also add this to your Meta App's Valid OAuth Redirect URIs:
     http://localhost:3000/api/auth/instagram/callback   ← for local dev
     https://yourdomain.com/api/auth/instagram/callback  ← for production


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  5. APIFY (Ethical Instagram scraping — competitor discovery)
     Get it at: https://apify.com
     → Settings → Integrations → API tokens
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  APIFY_API_TOKEN = apify_api_...


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  6. STRIPE (Payments — free/pro billing)
     Get it at: https://dashboard.stripe.com
     → Developers → API Keys
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  STRIPE_SECRET_KEY            = sk_test_...
  STRIPE_WEBHOOK_SECRET        = whsec_...   (from Stripe → Webhooks)
  STRIPE_PRO_MONTHLY_PRICE_ID  = price_...   (create in Stripe → Products)
  STRIPE_PRO_ANNUAL_PRICE_ID   = price_...   (create in Stripe → Products)

  ⚠ Register webhook in Stripe Dashboard → Webhooks:
     Endpoint URL: https://yourdomain.com/api/webhooks/stripe
     Events to listen for:
       - checkout.session.completed
       - customer.subscription.deleted

  PRIORITY ORDER — what to get first
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  To test core AI features locally (mock data OFF):
    1. OPENROUTER_API_KEY   ← most important, powers everything
    2. GOOGLE_API_KEY       ← Gemini video analysis

  To test competitor discovery:
    3. APIFY_API_TOKEN
    4. META_APP_ID + META_APP_SECRET

  To test payments:
    5. STRIPE keys

  To test RAG scripts:
    6. PINECONE keys

  ⭐ Right now with USE_MOCK_DATA=true you need NONE of these.
     The app runs fully on fixture data at http://localhost:3000
